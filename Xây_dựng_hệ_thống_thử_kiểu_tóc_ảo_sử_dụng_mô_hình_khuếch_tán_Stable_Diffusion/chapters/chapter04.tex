\section{Xây dựng Model AI (Giai đoạn huấn luyện)}
Ngoài việc sử dụng mô hình SDXL Inpainting có sẵn, hệ thống xây dựng quy trình huấn luyện hai giai đoạn nhằm cải thiện chất lượng sinh ảnh cho bài toán chuyên biệt.

\subsection{Giai đoạn 1: Huấn luyện bộ mã hóa kết cấu tóc (HairTextureEncoder)}
Bộ mã hóa sử dụng backbone ResNet\nobreakdash-50 để trích xuất vector đặc trưng tóc 2048 chiều. Quá trình huấn luyện sử dụng ảnh crop vùng tóc (xác định bằng SegFormer), resize về $128\times 128$ và chuẩn hóa theo ImageNet. Vector đặc trưng này phục vụ cho nhánh style conditioning trong quá trình suy luận.

\subsection{Giai đoạn 2: Tinh chỉnh UNet Inpainting bằng LoRA}
Mô hình UNet được mở rộng lên 13 kênh đầu vào để hỗ trợ \textbf{direct reference conditioning} -- cho phép mô hình trực tiếp nhận ảnh tóc mẫu làm điều kiện sinh ảnh thay vì chỉ dựa vào style embedding. Cấu trúc 13 kênh:
\begin{itemize}
    \item 4 kênh \textit{noisy latents}: Ảnh nhiễu ở bước khuếch tán hiện tại.
    \item 1 kênh \textit{mask}: Vùng tóc cần vẽ lại (binary mask downsampled).
    \item 4 kênh \textit{masked latents}: Ảnh gốc đã khoét vùng tóc (qua VAE Encoder).
    \item 4 kênh \textit{reference hair latents}: Ảnh tóc tham chiếu (qua VAE Encoder).
\end{itemize}

\subsubsection{Kiến trúc mạng}
\begin{itemize}
    \item \textbf{HairInpaintingUNet:} Wrapper mở rộng lớp \texttt{conv\_in} của SDXL từ 4/9 kênh lên 13 kênh. Trọng số cũ được giữ nguyên, 4 kênh mới khởi tạo bằng zero. Hệ thống tự động phát hiện và tương thích ngược với checkpoint 9-kênh.
    \item \textbf{CrossAttentionInjector:} Module inject style embedding (2048-d) và identity embedding (512-d) vào các tầng cross-attention của UNet thông qua linear projection riêng biệt.
\end{itemize}

\subsubsection{Chiến lược huấn luyện}
\begin{itemize}
    \item \textbf{Cross-Person Pairing:} 70\% mẫu huấn luyện ghép tóc từ \textit{người khác}, 30\% tự tham chiếu (self-reference). Chiến lược này buộc mô hình học cách tái tạo kiểu tóc từ ảnh mẫu bất kỳ thay vì chỉ ghi nhớ.
    \item \textbf{VAE Latent Caching:} Ảnh ground truth được pre-encode qua VAE Encoder thành latent $4 \times 64 \times 64$ và lưu cache trên đĩa (${\sim}32$KB/file). Cơ chế này giúp bỏ qua bước VAE encoding trong vòng lặp huấn luyện, tiết kiệm đáng kể VRAM và thời gian.
    \item \textbf{Mixed-Precision Training:} Lớp \texttt{conv\_in} (lớp duy nhất cần học trọng số mới) được giữ ở FP32 để đảm bảo gradient không bị underflow. Phần còn lại của UNet chạy ở FP16 thông qua \texttt{torch.amp.autocast}. Cơ chế \texttt{GradScaler} phóng đại loss trước khi lan truyền ngược rồi thu nhỏ lại, đảm bảo độ chính xác số học cho toàn pipeline.
    \item \textbf{Classifier-Free Guidance Dropout:} 10\% bước huấn luyện đặt toàn bộ conditioning (style, identity, reference latents) về zero, cho phép áp dụng CFG khi suy luận.
    \item \textbf{Quản lý bộ nhớ (OOM Prevention):} UNet 13-kênh tiêu tốn lượng VRAM lớn. Để tránh tràn bộ nhớ (Out-Of-Memory) trên GPU 15GB (Google Colab T4), khối phân tích (Monitor Block) sử dụng VGG16 và InceptionResnetV1 được cấu hình chạy \textit{sau} quá trình \texttt{backward()} nhằm giải phóng đồ thị tính toán đạo hàm (autograd graph) trước khi tải thêm mô hình vào VRAM. Kỹ thuật này tái sử dụng $\sim$3-4GB VRAM bị chiếm dụng. Bên cạnh đó, hệ thống bổ sung cơ chế kiểm soát nhịp đập bộ nhớ (Heartbeat Logging) mỗi 100 bước để theo dõi tình trạng phân bổ RAM.
    \item \textbf{Tối ưu hóa I/O cục bộ (Auto-Caching):} Quá trình truy xuất dữ liệu trên ổ đĩa mạng lưới (Google Drive FUSE) thường xuyên gây nghẽn cổ chai (bottleneck) và làm treo luồng DataLoader. Hệ thống áp dụng cơ chế Caching phần cứng hai chiều (Two-Way Sync):
    \begin{enumerate}
        \item Từng cụm dữ liệu phân mảnh (Chunk) dung lượng $\sim$6.7GB được sao chép hoàn toàn sang ổ đĩa trạng thái rắn (SSD) siêu tốc đặt tại \texttt{/tmp/} của máy chủ trước khi đọc. Điều này loại bỏ hoàn toàn độ trễ mạng, giảm thời gian giải mã VAE xuống $\sim$15 lần (từ 45 phút xuống dưới 3 phút).
        \item Hệ thống sau đó quét trên SSD, chỉ trích xuất các tệp latent còn thiếu và đồng bộ tức thì thư mục \texttt{gt\_latents\_cache} về lại lưu trữ đám mây. Kiến trúc "Fail-Safe" này đảm bảo dữ liệu quá trình không bị mất ngay cả khi máy chủ ảo đột ngột sập nguồn. DataLoader cũng được ép buộc dùng luồng đơn (\texttt{num\_workers=0}) để tối đa hóa tính ổn định.
    \end{enumerate}
\end{itemize}

\subsubsection{Hàm mất mát}
\begin{itemize}
    \item \textbf{Mask-aware loss:} Tăng trọng số cho pixel vùng tóc và biên tóc.
    \item \textbf{Latent loss:} So sánh đặc trưng trong không gian tiềm ẩn.
    \item \textbf{Cơ chế weighting theo timestep:} Hỗ trợ hội tụ ổn định.
\end{itemize}

Quá trình huấn luyện sử dụng LoRA rank 16, độ phân giải $512\times 512$, batch size 2 với gradient accumulation 4 bước. Chỉ khoảng 35 triệu tham số (${\sim}1.3$\% tổng mô hình) được huấn luyện, phù hợp với tài nguyên GPU 12--16GB (RTX 3060 / Colab T4).

\section{Xây dựng Backend API}
Backend được xây dựng bằng FastAPI với cấu trúc module rõ ràng.

\subsection{Cấu trúc source code}
\begin{Verbatim}[frame=single]
backend/app/
|-- main.py          # FastAPI endpoints + CORS + Static Files
|-- tasks.py         # 3 Celery tasks (hair_transfer, colorize, detect_faces)
|-- config.py        # Model paths + App settings
|-- schemas.py       # Pydantic request/response schemas
|-- services/
|   |-- diffusion.py       # Dual pipeline (Standard + Custom)
|   |-- face.py            # YOLO + InsightFace + AdaFace
|   |-- mask.py            # SegFormer + Dynamic Masking
|   |-- hair_color_service.py  # HSV Color Transfer
|   |-- embedder.py        # Yaw-based embedding switching
|   |-- face_detector.py   # YOLOv8-Face + NMS
|   |-- pose_estimator.py  # InsightFace landmarks + 3DDFA pose
|   \-- reconstructor_3d.py   # 3DDFA V2 dense mesh
\-- utils/
    \-- torch_patch.py     # PyTorch compatibility patches
\end{Verbatim}

\subsection{Celery Tasks và Lazy Loading}
Hệ thống định nghĩa ba Celery task chính:
\begin{enumerate}
    \item \textbf{process\_hair\_transfer:} Pipeline đầy đủ gồm load ảnh, phân tích khuôn mặt (auto-rotate nếu không phát hiện mặt), phân đoạn tóc, mở rộng mask động, ước lượng depth, suy luận Diffusion, đổi màu tóc (nếu yêu cầu) và lưu kết quả.
    \item \textbf{process\_hair\_colorize:} Chỉ cần hair mask + HSV color transfer, nhanh hơn đáng kể so với pipeline đầy đủ.
    \item \textbf{process\_detect\_faces:} Quét khuôn mặt, mở rộng bounding box 1.8$\times$ để lấy trọn tóc và cổ, cắt và lưu ảnh crop riêng.
\end{enumerate}

Tất cả model AI được lazy-load trong worker process qua hàm \texttt{get\_services()} để đảm bảo tương thích CUDA/Celery. Sau mỗi task, bộ nhớ GPU được giải phóng bằng \texttt{torch.cuda.empty\_cache()}.

\subsection{Pydantic Schemas}
Dữ liệu đầu vào và đầu ra được kiểm tra bằng Pydantic:
\begin{itemize}
    \item \textbf{HairTransferRequest/Response:} Ảnh đầu vào, prompt, màu tóc tùy chọn (preset hoặc hex), cường độ màu (0.0--1.0).
    \item \textbf{TaskStatusResponse:} Trạng thái PENDING/PROCESSING/SUCCESS/FAILURE, URL kết quả, danh sách faces (cho detect), thông báo lỗi.
    \item \textbf{DetectFacesResponse:} Danh sách khuôn mặt (face\_id, bbox, confidence, cropped\_image\_url).
\end{itemize}

    \end{tabularx}

Giao diện sử dụng module \texttt{hairApi.js} để giao tiếp với backend thông qua 5 hàm API:
\begin{itemize}
    \item \texttt{generateHair()}: Gửi 2 ảnh + prompt + màu tóc qua \texttt{POST /generate}.
    \item \texttt{detectFaces()}: Gửi ảnh để quét khuôn mặt qua \texttt{POST /detect-faces}.
    \item \texttt{pollTask()}: Poll trạng thái task mỗi 2 giây cho đến khi hoàn thành hoặc thất bại.
    \item \texttt{getColors()}: Lấy danh sách preset màu.
    \item \texttt{getRandomPair()}: Lấy cặp ảnh FFHQ ngẫu nhiên để test.
\end{itemize}

Luồng xử lý trên frontend khi người dùng nhấn "Vẽ":
\begin{enumerate}
    \item Gọi \texttt{/detect-faces} trên ảnh chân dung. Nếu phát hiện nhiều khuôn mặt, hiện FaceSelector để chọn.
    \item Gọi \texttt{/detect-faces} trên ảnh tóc tham chiếu. Tương tự, chọn khuôn mặt nếu cần.
    \item Gọi \texttt{/generate} với ảnh đã crop + prompt + màu tóc.
    \item Poll trạng thái, hiển thị kết quả khi hoàn thành.
\end{enumerate}

\section{Triển khai bằng Docker}
Hệ thống được container hóa qua Docker Compose với 4 service:
\begin{enumerate}
    \item \textbf{Redis} (\texttt{redis:7-alpine}): Message broker với healthcheck.
    \item \textbf{Backend} (\texttt{tryhair\_backend}): FastAPI server, mount volume \texttt{models/} và \texttt{data/}, GPU passthrough.
    \item \textbf{Celery Worker} (\texttt{tryhair\_celery}): Dùng chung Dockerfile với backend, chạy lệnh Celery worker với \texttt{--pool=solo --concurrency=1} cho GPU.
    \item \textbf{Frontend} (\texttt{tryhair\_frontend}): Build React thành static files, serve qua Nginx trên port 3000.
\end{enumerate}
