\section{Kết quả đạt được}
Đề tài đã xây dựng hệ thống \textbf{TryHairStyle} có khả năng tiếp nhận ảnh người dùng và ảnh kiểu tóc tham chiếu để tạo ảnh thử kiểu tóc trên nền tảng web. Các kết quả chính gồm:
\begin{itemize}
    \item \textbf{Pipeline xử lý tích hợp:} Hệ thống tích hợp các mô-đun phân tích khuôn mặt, tạo mặt nạ, ước lượng pose, trích xuất đặc trưng, tái dựng hình học và đổi màu tóc vào một quy trình xử lý thống nhất.
    \item \textbf{Cơ chế điều kiện kép:} Nhánh bảo toàn danh tính (InsightFace/AdaFace + ControlNet Depth) và nhánh truyền tải đặc trưng kiểu tóc (IP-Adapter Plus/HairTextureEncoder) được kết hợp trong cùng quá trình suy luận để duy trì cấu trúc khuôn mặt và kiểu tóc tham chiếu.
    \item \textbf{Bảo toàn bối cảnh nền:} Hệ thống sử dụng tọa độ \texttt{bbox} để dán ngược vùng ảnh kết quả vào ảnh gốc, qua đó hạn chế thay đổi đối với vùng nền ngoài khu vực chỉnh sửa.
    \item \textbf{Cơ chế cách ly tiến trình suy luận:} Khối thuật toán AI được khởi chạy trong tiến trình cách ly (\textit{Subprocess}), góp phần giảm rủi ro treo hệ thống do tràn bộ nhớ dùng chung.
    \item \textbf{Tối ưu tài nguyên tính toán:} Hệ thống áp dụng các kỹ thuật quản lý VRAM và \textit{Auto-Caching} để duy trì quá trình huấn luyện ổn định trên môi trường Google Colab T4 15GB.
    \item \textbf{Mặt nạ động thích ứng:} Cơ chế mở rộng mặt nạ tóc được thiết kế dựa trên tương quan giữa tóc tham chiếu và tóc gốc nhằm hỗ trợ các trường hợp tóc dài, tóc phồng hoặc ít tóc.
    \item \textbf{Chức năng giao diện hỗ trợ thao tác:} Giao diện cung cấp các chức năng như đổi màu tóc nhanh theo HSV, dịch prompt Việt--Anh tự động và làm sạch dữ liệu nhập bằng nút \textit{Clear}.
    \item \textbf{Kiến trúc xử lý bất đồng bộ:} Hệ thống sử dụng FastAPI, Celery và Redis để tiếp nhận, xếp hàng và xử lý tác vụ sinh ảnh theo cơ chế bất đồng bộ; đồng thời hỗ trợ triển khai bằng Docker Compose.
    \item \textbf{Quy trình benchmark ngoại tuyến cho checkpoint:} Hệ thống xây dựng bộ \textit{eval set} cố định 30 mẫu, quy trình sinh ảnh thật cho từng checkpoint và cơ chế tổng hợp các chỉ số \textit{identity}, \textit{hairstyle}, \textit{background}, \textit{naturalness}, \textit{runtime/stability} để hỗ trợ lựa chọn mô hình.
\end{itemize}

\section{Đánh giá ưu và nhược điểm hệ thống}

\subsection{Ưu điểm nổi bật}
\begin{itemize}
    \item \textbf{Chất lượng sinh ảnh và bảo toàn nhân dạng cao:} Benchmark hiện tại cho \textit{identity similarity} trung bình 0.9517, cho thấy mô hình giữ được phần lớn đặc trưng khuôn mặt trên bộ eval set 30 mẫu.
    \item \textbf{Bảo toàn bối cảnh nền:} Chỉ số \textit{background PSNR} trung bình đạt 20.29 dB và \textit{background SSIM} đạt 0.8948. Các kết quả này phù hợp với cơ chế dán ngược theo \texttt{bbox}, giúp hạn chế biến đổi ở phần nền ngoài vùng tóc.
    \item \textbf{Vận hành ổn định trên cấu hình phần cứng hạn chế:} Cơ chế hàng đợi Redis kết hợp Celery giúp hệ thống tiếp nhận và xử lý yêu cầu theo hướng bất đồng bộ, qua đó giảm nguy cơ treo giao diện khi khối lượng tác vụ tăng.
    \item \textbf{Giao diện hỗ trợ nhiều chức năng thao tác:} Frontend hỗ trợ dịch prompt Việt--Anh tự động, đổi màu tóc nhanh và lựa chọn mô hình AI, góp phần thuận tiện hóa quá trình sử dụng.
    \item \textbf{Đã có quy trình đánh giá lặp lại được:} Việc bổ sung \textit{eval set}, ảnh sinh thật và bảng benchmark giúp phần đánh giá bớt phụ thuộc vào cảm quan, đồng thời hỗ trợ tốt hơn cho việc chọn checkpoint sử dụng trong hệ thống. Ở lần benchmark hiện tại, checkpoint \texttt{deep\_hair\_v1} đạt \textit{final score} 0.5768.
\end{itemize}

\subsection{Hạn chế tồn đọng}
\begin{itemize}
    \item \textbf{Độ trễ xử lý còn cao:} Thời gian suy luận vài giây mỗi yêu cầu trên GPU 12GB, chưa phù hợp thời gian thực.
    \item \textbf{Yêu cầu phần cứng lớn:} Cần GPU NVIDIA có VRAM tối thiểu 12GB để nạp đồng thời nhiều model.
    \item \textbf{Phụ thuộc chất lượng ảnh đầu vào:} Ảnh quá mờ, che khuất nhiều hoặc ánh sáng phức tạp có thể ảnh hưởng chất lượng mask và ảnh sinh ra.
    \item \textbf{Vùng biên tóc chưa hoàn hảo:} Một số trường hợp vẫn lộ viền hoặc chuyển tiếp chưa tự nhiên tại ranh giới tóc--da.
    \item \textbf{Chất lượng ảnh sinh ra chưa ổn định:} Mặc dù trong một số trường hợp hệ thống tạo ra kết quả có chất lượng tốt, vẫn tồn tại nhiều trường hợp ảnh đầu ra không đạt yêu cầu sử dụng.
    \item \textbf{Chỉ số bám kiểu tóc vẫn còn khó:} Sau khi chuyển sang so sánh tập trung trên vùng tóc đã tách riêng, nhóm \textit{hairstyle} đã cải thiện lên mức 0.1471, tuy nhiên vẫn thấp hơn kỳ vọng và chưa phản ánh trọn vẹn cảm nhận thị giác của người dùng trong các trường hợp tóc dài, tóc mái hoặc tóc có độ phồng lớn.
\end{itemize}

\section{Hướng phát triển}
\begin{itemize}
    \item \textbf{Nâng cao chất lượng mô hình:} Tăng số lượng epoch huấn luyện nhằm cải thiện khả năng tổng quát hóa và chất lượng ảnh sinh ra.
    
    \item \textbf{Tối ưu tài nguyên tính toán:} Nghiên cứu các kỹ thuật tối ưu nhằm giảm yêu cầu phần cứng, từ đó giúp hệ thống có thể triển khai rộng rãi trên nhiều nền tảng hơn.
    
    \item \textbf{Tăng tốc suy luận:} Áp dụng các phương pháp như giảm số bước khuếch tán, distillation, hoặc sử dụng các kiến trúc suy luận nhanh (LCM, Turbo) để rút ngắn thời gian xử lý.
    
    \item \textbf{Mở rộng sang video và thời gian thực:} Phát triển khả năng xử lý video hoặc thử kiểu tóc trực tiếp trên webcam, hướng đến các ứng dụng tương tác thời gian thực.
    
    \item \textbf{Chuẩn hóa đánh giá ở mức sâu hơn:} Tiếp tục hoàn thiện bộ benchmark đã xây dựng bằng cách nâng cấp metric cho \textit{hairstyle similarity}, thử thêm các tiêu chí như \textit{hair-mask overlap} hoặc đặc trưng học sâu trên vùng tóc, đồng thời mở rộng thêm nhiều checkpoint để việc chọn mô hình có độ tin cậy cao hơn.
    
    \item \textbf{Triển khai trên hạ tầng đám mây:} Xây dựng hệ thống trên nền tảng điện toán đám mây nhằm đáp ứng nhu cầu truy cập đồng thời của nhiều người dùng.
    
    \item \textbf{Mở rộng nền tảng ứng dụng:} Phát triển ứng dụng di động hoặc Progressive Web App (PWA) để nâng cao khả năng tiếp cận và trải nghiệm người dùng.
\end{itemize}
