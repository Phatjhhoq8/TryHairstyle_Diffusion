\section{Kết quả đạt được}
Đề tài đã xây dựng được hệ thống \textbf{TryHairStyle} có khả năng tiếp nhận ảnh người dùng và ảnh kiểu tóc tham chiếu để tạo ảnh thử kiểu tóc trên nền tảng web. Các kết quả nổi bật gồm:
\begin{itemize}
    \item \textbf{Pipeline AI hoàn thiện phối hợp đa luồng:} Tích hợp liền mạch hệ module lõi phân tích hình thái khuôn mặt (Face, Mask, Pose, Embedder, Reconstructor, Color) thành một quy trình xử lý liên hoàn tự động từ đầu đến cuối.
    \item \textbf{Cơ chế điều kiện kép:} Nhánh cấu trúc bản dạng (InsightFace/AdaFace + ControlNet Depth) và nhánh chuyển giao giao diện kiểu tóc (IP-Adapter Plus/HairTextureEncoder) chạy song song, giúp hợp nhất tỷ lệ hoàn hảo.
    \item \textbf{Bảo toàn bối cảnh (Background Preservation):} Áp dụng giải thuật định vị tọa độ \texttt{bbox} chuẩn trước khi suy luận và chèn đè khuôn mặt kết quả về ảnh gốc, giữ không gian môi trường của người dùng không biến đổi.
    \item \textbf{Quản trị linh hoạt bộ nhớ tự sinh (Microservices):} Cơ chế khởi chạy nền thuật toán AI đồ sộ bằng tiến trình cách ly (\textit{Subprocess}) giúp hạn chế rủi ro treo hệ thống do tràn bộ nhớ chia sẻ chung. 
    \item \textbf{Tối ưu hóa tài nguyên phần cứng:} Vận dụng các kỹ thuật quản lý VRAM (tái cấu trúc đồ thị đạo hàm) và bộ truy xuất lưu trữ tĩnh (I/O Auto-Caching), giữ tính ổn định vững vàng mô hình học sâu khổng lồ trong suốt quá trình huấn luyện trên máy chủ đám mây (Google Colab T4 15GB).
    \item \textbf{Mặt nạ động nội suy:} Cơ chế tính khối lượng diện tích mở rộng dựa trên so sánh mẫu và ảnh gốc.
    \item \textbf{Trải nghiệm tiện ích Giao diện mở rộng:} 8 thư viện React thành phần tối ưu thao tác bằng một cú nhấp chuột đa chức năng (Đổi màu tóc tức thời hệ HSV, Dịch cú pháp Prompt Việt-Anh tự động, thanh trượt dọn rác giao diện bằng nút Clear).
    \item \textbf{Kiến trúc bất đồng bộ \& Microservices:} FastAPI + Celery + Redis xử lý tác vụ không đồng bộ; kết hợp cơ chế triển khai vùng chứa (Container) Docker Compose trích xuất kết nối liên GPU passthrough toàn diện.
    \item \textbf{Bổ sung benchmark ngoại tuyến cho checkpoint:} Hệ thống đã có bộ \textit{eval set} cố định 30 mẫu, quy trình sinh ảnh thật cho từng checkpoint và cơ chế tổng hợp các chỉ số \textit{identity}, \textit{hairstyle}, \textit{background}, \textit{naturalness}, \textit{runtime/stability} để hỗ trợ chọn mô hình đáng tin cậy hơn.
\end{itemize}

\section{Đánh giá ưu và nhược điểm hệ thống}

\subsection{Ưu điểm nổi bật}
\begin{itemize}
    \item \textbf{Chất lượng sinh ảnh và bảo toàn nhân dạng cao:} Benchmark hiện tại cho \textit{identity similarity} trung bình 0.9517, cho thấy mô hình giữ được phần lớn đặc trưng khuôn mặt trên bộ eval set 30 mẫu.
    \item \textbf{Bảo toàn bối cảnh nền ở mức khá:} Chỉ số \textit{background PSNR} trung bình đạt 20.29 dB và \textit{background SSIM} đạt 0.8948. Điều này phù hợp với cơ chế dán ngược theo \texttt{bbox}, giúp phần nền ngoài vùng tóc ít bị biến đổi đáng kể.
    \item \textbf{Phục vụ trơn tru ở phần cứng eo hẹp:} Các thiết lập hàng đợi message broker (Redis) kết hợp xử lý ngắt nhịp Celery giúp loại bỏ lỗi treo ứng dụng khi có yêu cầu cường độ cao trong UI.
    \item \textbf{Tùy biến cực mạnh giao diện Front-end:} Tính năng phiên dịch nội hàm Việt-Anh tự động, Đổi sắc thái màu nhanh và chọn model AI từ popup mang đến tính phổ cập tương tác. Thiết lập container bằng Docker Compose đưa toàn bộ cấu hình phức tạp ẩn mình.
    \item \textbf{Đã có quy trình đánh giá lặp lại được:} Việc bổ sung \textit{eval set}, ảnh sinh thật và bảng benchmark giúp phần đánh giá bớt phụ thuộc vào cảm quan, đồng thời hỗ trợ tốt hơn cho việc chọn checkpoint sử dụng trong hệ thống. Ở lần benchmark hiện tại, checkpoint \texttt{deep\_hair\_v1} đạt \textit{final score} 0.5768.
\end{itemize}

\subsection{Hạn chế tồn đọng}
\begin{itemize}
    \item \textbf{Độ trễ xử lý còn cao:} Thời gian suy luận vài giây mỗi yêu cầu trên GPU 12GB, chưa phù hợp thời gian thực.
    \item \textbf{Yêu cầu phần cứng lớn:} Cần GPU NVIDIA có VRAM tối thiểu 12GB để nạp đồng thời nhiều model.
    \item \textbf{Phụ thuộc chất lượng ảnh đầu vào:} Ảnh quá mờ, che khuất nhiều hoặc ánh sáng phức tạp có thể ảnh hưởng chất lượng mask và ảnh sinh ra.
    \item \textbf{Vùng biên tóc chưa hoàn hảo:} Một số trường hợp vẫn lộ viền hoặc chuyển tiếp chưa tự nhiên tại ranh giới tóc--da.
    \item \textbf{Chất lượng ảnh sinh ra chưa ổn định:} Mặc dù trong một số trường hợp hệ thống tạo ra kết quả có chất lượng tốt, vẫn tồn tại nhiều trường hợp ảnh đầu ra không đạt yêu cầu sử dụng.
    \item \textbf{Chỉ số bám kiểu tóc vẫn còn khó:} Sau khi chuyển sang so sánh tập trung trên vùng tóc đã tách riêng, nhóm \textit{hairstyle} đã cải thiện lên mức 0.1471, tuy nhiên vẫn thấp hơn kỳ vọng và chưa phản ánh trọn vẹn cảm nhận thị giác của người dùng trong các trường hợp tóc dài, tóc mái hoặc tóc có độ phồng lớn.
\end{itemize}

\section{Hướng phát triển}
\begin{itemize}
    \item \textbf{Nâng cao chất lượng mô hình:} Tăng số lượng epoch huấn luyện nhằm cải thiện khả năng tổng quát hóa và chất lượng ảnh sinh ra.
    
    \item \textbf{Tối ưu tài nguyên tính toán:} Nghiên cứu các kỹ thuật tối ưu nhằm giảm yêu cầu phần cứng, từ đó giúp hệ thống có thể triển khai rộng rãi trên nhiều nền tảng hơn.
    
    \item \textbf{Tăng tốc suy luận:} Áp dụng các phương pháp như giảm số bước khuếch tán, distillation, hoặc sử dụng các kiến trúc suy luận nhanh (LCM, Turbo) để rút ngắn thời gian xử lý.
    
    \item \textbf{Mở rộng sang video và thời gian thực:} Phát triển khả năng xử lý video hoặc thử kiểu tóc trực tiếp trên webcam, hướng đến các ứng dụng tương tác thời gian thực.
    
    \item \textbf{Chuẩn hóa đánh giá ở mức sâu hơn:} Tiếp tục hoàn thiện bộ benchmark đã xây dựng bằng cách nâng cấp metric cho \textit{hairstyle similarity}, thử thêm các tiêu chí như \textit{hair-mask overlap} hoặc đặc trưng học sâu trên vùng tóc, đồng thời mở rộng thêm nhiều checkpoint để việc chọn mô hình có độ tin cậy cao hơn.
    
    \item \textbf{Triển khai trên hạ tầng đám mây:} Xây dựng hệ thống trên nền tảng điện toán đám mây nhằm đáp ứng nhu cầu truy cập đồng thời của nhiều người dùng.
    
    \item \textbf{Mở rộng nền tảng ứng dụng:} Phát triển ứng dụng di động hoặc Progressive Web App (PWA) để nâng cao khả năng tiếp cận và trải nghiệm người dùng.
\end{itemize}
